224/* Check if current SBI specification version is 0.1 or not */
225static inline int sbi_spec_is_0_1(void)
226{
227 return (sbi_spec_version == SBI_SPEC_VERSION_DEFAULT) ? 1 : 0;
228}
229
230/* Get the major version of SBI */
231static inline unsigned long sbi_major_version(void)
232{
233 return (sbi_spec_version >> SBI_SPEC_VERSION_MAJOR_SHIFT) &
234 SBI_SPEC_VERSION_MAJOR_MASK;
235}
236
237/* Get the minor version of SBI */
238static inline unsigned long sbi_minor_version(void)
239{
240 return sbi_spec_version & SBI_SPEC_VERSION_MINOR_MASK;
241}
242
243/* Make SBI version */
244static inline unsigned long sbi_mk_version(unsigned long major,
245 unsigned long minor)
246{
247 return ((major & SBI_SPEC_VERSION_MAJOR_MASK)
248 << SBI_SPEC_VERSION_MAJOR_SHIFT) |
249 minor;
250}

254static inline struct sbiret sbi_ecall(int ext, int fid, unsigned long arg0,
255 unsigned long arg1, unsigned long arg2,
256 unsigned long arg3, unsigned long arg4,
257 unsigned long arg5)
258{
259 struct sbiret ret;
260
261 register uintptr_t a0 asm("a0") = (uintptr_t)(arg0);
262 register uintptr_t a1 asm("a1") = (uintptr_t)(arg1);
263 register uintptr_t a2 asm("a2") = (uintptr_t)(arg2);
264 register uintptr_t a3 asm("a3") = (uintptr_t)(arg3);
265 register uintptr_t a4 asm("a4") = (uintptr_t)(arg4);
266 register uintptr_t a5 asm("a5") = (uintptr_t)(arg5);
267 register uintptr_t a6 asm("a6") = (uintptr_t)(fid);
268 register uintptr_t a7 asm("a7") = (uintptr_t)(ext);
269 asm volatile("ecall"
270 : "+r"(a0), "+r"(a1)
271 : "r"(a2), "r"(a3), "r"(a4), "r"(a5), "r"(a6), "r"(a7)
272 : "memory");
273 ret.error = a0;
274 ret.value = a1;
275
276 return ret;
277}

279static inline long __sbi_base_ecall(int fid)
280{
281 struct sbiret ret;
282
283 ret = sbi_ecall(SBI_EXT_BASE, fid, 0, 0, 0, 0, 0, 0);
284 if (!ret.error)
285 return ret.value;
286 else
287 return 0;
288}
